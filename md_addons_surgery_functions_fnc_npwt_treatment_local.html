private _bandagedWounds = GET_BANDAGED_WOUNDS(_patient); private _stitchedWounds = GET_STITCHED_WOUNDS(_patient); private _openWounds = GET_OPEN_WOUNDS(_patient); private _bandaged = []; private _stitched = []; private _open = [];

if (_lidocaine == 0 && _morphine == 0) then { private _pain = random [0.7, 0.8, 0.9]; [_patient, _pain] call ACEFUNC(medical_status,adjustPainLevel); };

{ _x params ["_id", "_bodyPart", "_amount"];

} forEach _openWounds;

} forEach _openWounds;
